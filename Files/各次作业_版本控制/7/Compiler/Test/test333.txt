char mytoupper(char inputchar)
{
    if (inputchar == 'a')
    {
        return ('a');
    }
    if (inputchar == 'b')
        return ('b');
    if (inputchar == 'c')
    {
        return ('c');
    }
    if (inputchar == 'd')
    {
        return ('d');
    }
    if (inputchar == 'e')
        return ('e');
}
